# TTS Stack Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a standalone GitHub deployment package that exposes an OpenAI-compatible TTS API through a Go gateway in front of `mzzsfy/tts`, with manual multi-token YAML configuration, per-token defaults, and hot-reloadable business policy.

**Architecture:** The repository contains a Go `tts-gateway` application plus a self-contained deployment package made of `compose.yaml`, config templates, scripts, and release automation. `tts-gateway` is the only public service; it validates bearer tokens, enforces per-token policy, and proxies synthesis and voice-list requests to a private `mzzsfy/tts` container.

**Tech Stack:** Go 1.24, `net/http`, `log/slog`, `gopkg.in/yaml.v3`, `github.com/fsnotify/fsnotify`, Docker Compose, GitHub Actions, GHCR

### Design constraints this plan must preserve

- `POST /v1/audio/speech` remains OpenAI-compatible in request shape
- Gateway only accepts `model`, `input`, and optional `voice`
- Output is always MP3 stream
- Callers cannot set `thread`, `shardLength`, `speed`, or `response_format`
- Config hot reload applies only to token/business policy
- Upstream base URL and listen port are not hot reloaded
- No caching, no rate limiting, no audio file persistence, no admin UI

```go
package config

import (
	"fmt"
	"os"

	"gopkg.in/yaml.v3"
)

type RuntimeConfig struct {
	Server   ServerConfig   `yaml:"server"`
	Upstream UpstreamConfig `yaml:"upstream"`
	Defaults Defaults       `yaml:"defaults"`
	Tokens   []TokenConfig  `yaml:"tokens"`
}

type ServerConfig struct {
	Port int `yaml:"port"`
}

type UpstreamConfig struct {
	BaseURL        string `yaml:"base_url"`
	TimeoutSeconds int    `yaml:"timeout_seconds"`
}

type Defaults struct {
	Voice         string `yaml:"voice"`
	Thread        int    `yaml:"thread"`
	ShardLength   int    `yaml:"shard_length"`
	MaxTextLength int    `yaml:"max_text_length"`
}

type TokenConfig struct {
	Name          string   `yaml:"name"`
	Token         string   `yaml:"token"`
	Enabled       bool     `yaml:"enabled"`
	Defaults      Defaults `yaml:"defaults"`
	AllowedVoices []string `yaml:"allowed_voices"`
}

func Load(path string) (*RuntimeConfig, error) {
	body, err := os.ReadFile(path)
	if err != nil {
		return nil, fmt.Errorf("read config: %w", err)
	}

	var cfg RuntimeConfig
	if err := yaml.Unmarshal(body, &cfg); err != nil {
		return nil, fmt.Errorf("decode yaml: %w", err)
	}

	if err := cfg.Validate(); err != nil {
		return nil, err
	}

	return &cfg, nil
}

func (c *RuntimeConfig) Validate() error {
	if c.Server.Port <= 0 {
		return fmt.Errorf("server.port must be greater than zero")
	}

	if c.Upstream.BaseURL == "" {
		return fmt.Errorf("upstream.base_url is required")
	}

	if c.Upstream.TimeoutSeconds <= 0 {
		return fmt.Errorf("upstream.timeout_seconds must be greater than zero")
	}

	if c.Defaults.Thread <= 0 {
		return fmt.Errorf("defaults.thread must be greater than zero")
	}

	if c.Defaults.ShardLength <= 0 {
		return fmt.Errorf("defaults.shard_length must be greater than zero")
	}

	if c.Defaults.MaxTextLength <= 0 {
		return fmt.Errorf("defaults.max_text_length must be greater than zero")
	}

	if len(c.Tokens) == 0 {
		return fmt.Errorf("at least one token is required")
	}

	for i, token := range c.Tokens {
		if token.Name == "" {
			return fmt.Errorf("tokens[%d].name is required", i)
		}
		if token.Token == "" {
			return fmt.Errorf("tokens[%d].token is required", i)
		}
		if token.Defaults.Voice == "" {
			return fmt.Errorf("tokens[%d].defaults.voice is required", i)
		}
		if token.Defaults.Thread <= 0 {
			return fmt.Errorf("tokens[%d].defaults.thread must be greater than zero", i)
		}
		if token.Defaults.ShardLength <= 0 {
			return fmt.Errorf("tokens[%d].defaults.shard_length must be greater than zero", i)
		}
		if token.Defaults.MaxTextLength <= 0 {
			return fmt.Errorf("tokens[%d].defaults.max_text_length must be greater than zero", i)
		}
		if len(token.AllowedVoices) == 0 {
			return fmt.Errorf("tokens[%d].allowed_voices must contain at least one item", i)
		}
	}

	return nil
}
```

```go
package config

import (
	"fmt"
	"sync/atomic"
)

type Store struct {
	path string
	ptr  atomic.Pointer[RuntimeConfig]
}

func NewStore(path string) (*Store, error) {
	cfg, err := Load(path)
	if err != nil {
		return nil, err
	}

	store := &Store{path: path}
	store.ptr.Store(cfg)
	return store, nil
}

func (s *Store) Current() *RuntimeConfig {
	return s.ptr.Load()
}

func (s *Store) Reload() error {
	cfg, err := Load(s.path)
	if err != nil {
		return fmt.Errorf("reload config: %w", err)
	}

	s.ptr.Store(cfg)
	return nil
}
```

Create `gateway/internal/config/watch.go`:

```go
package config

import (
	"context"
	"log/slog"
	"path/filepath"

	"github.com/fsnotify/fsnotify"
)

func Watch(ctx context.Context, logger *slog.Logger, store *Store, path string) error {
	watcher, err := fsnotify.NewWatcher()
	if err != nil {
		return err
	}

	go func() {
		defer watcher.Close()

		for {
			select {
			case <-ctx.Done():
				return
			case event, ok := <-watcher.Events:
				if !ok {
					return
				}
				if filepath.Clean(event.Name) != filepath.Clean(path) {
					continue
				}
				if event.Op&(fsnotify.Write|fsnotify.Create|fsnotify.Rename) == 0 {
					continue
				}
				if err := store.Reload(); err != nil {
					logger.Error("config reload failed", "error", err)
					continue
				}
				logger.Info("config reloaded", "path", path)
			case err, ok := <-watcher.Errors:
				if !ok {
					return
				}
				logger.Error("config watcher error", "error", err)
			}
		}
	}()

	return watcher.Add(filepath.Dir(path))
}
```

```go
package upstream

import (
	"context"
	"encoding/json"
	"fmt"
	"net/http"
	"net/url"
	"path"
	"strconv"
	"time"
)

type Client struct {
	baseURL string
	http    *http.Client
}

func New(baseURL string, timeoutSeconds int) *Client {
	return &Client{
		baseURL: baseURL,
		http: &http.Client{
			Timeout: time.Duration(timeoutSeconds) * time.Second,
		},
	}
}

func (c *Client) ListVoices(ctx context.Context) ([]Voice, error) {
	endpoint, err := url.Parse(c.baseURL)
	if err != nil {
		return nil, fmt.Errorf("parse upstream url: %w", err)
	}
	endpoint.Path = path.Join(endpoint.Path, "/api/tts/list")

	req, err := http.NewRequestWithContext(ctx, http.MethodGet, endpoint.String(), nil)
	if err != nil {
		return nil, fmt.Errorf("build voices request: %w", err)
	}

	resp, err := c.http.Do(req)
	if err != nil {
		return nil, fmt.Errorf("list voices request: %w", err)
	}
	defer resp.Body.Close()

	if resp.StatusCode != http.StatusOK {
		return nil, fmt.Errorf("list voices returned status %d", resp.StatusCode)
	}

	var voices []Voice
	if err := json.NewDecoder(resp.Body).Decode(&voices); err != nil {
		return nil, fmt.Errorf("decode voices: %w", err)
	}

	return voices, nil
}

func (c *Client) Synthesize(ctx context.Context, params SynthesizeParams) (*http.Response, error) {
	endpoint, err := url.Parse(c.baseURL)
	if err != nil {
		return nil, fmt.Errorf("parse upstream url: %w", err)
	}
	endpoint.Path = path.Join(endpoint.Path, "/api/tts")

	values := endpoint.Query()
	values.Set("text", params.Text)
	values.Set("voiceName", params.Voice)
	values.Set("thread", strconv.Itoa(params.Thread))
	values.Set("shardLength", strconv.Itoa(params.ShardLength))
	endpoint.RawQuery = values.Encode()

	req, err := http.NewRequestWithContext(ctx, http.MethodGet, endpoint.String(), nil)
	if err != nil {
		return nil, fmt.Errorf("build synthesize request: %w", err)
	}

	resp, err := c.http.Do(req)
	if err != nil {
		return nil, fmt.Errorf("synthesize request: %w", err)
	}

	return resp, nil
}
```

```go
type errorEnvelope struct {

type errorBody struct {

	Message string `json:"message"`

func writeError(w http.ResponseWriter, status int, typ, message string) {
	w.Header().Set("Content-Type", "application/json")
	w.WriteHeader(status)
	_ = json.NewEncoder(w).Encode(errorEnvelope{
		Error: errorBody{
			Type:    typ,
			Message: message,
		},
	})
```

```go
package httpapi

import (
	"io"
	"net/http"
	"strings"

	"github.com/songtf/tts-stack/gateway/internal/config"
	"github.com/songtf/tts-stack/gateway/internal/upstream"
)

type Upstream interface {
	ListVoices(ctx context.Context) ([]upstream.Voice, error)
	Synthesize(ctx context.Context, params upstream.SynthesizeParams) (*http.Response, error)
}

type App struct {
	cfg      *config.RuntimeConfig
	upstream Upstream
	mux      *http.ServeMux
}

func New(cfg *config.RuntimeConfig, upstreamClient Upstream) *App {
	app := &App{
		cfg:      cfg,
		upstream: upstreamClient,
		mux:      http.NewServeMux(),
	}

	app.mux.HandleFunc("GET /healthz", app.handleHealthz)
	app.mux.HandleFunc("GET /api/voices", app.handleVoices)
	app.mux.HandleFunc("POST /v1/audio/speech", app.handleSpeech)
	return app
}

func (a *App) ServeHTTP(w http.ResponseWriter, r *http.Request) {
	a.mux.ServeHTTP(w, r)
}

func (a *App) findToken(raw string) (*config.TokenConfig, bool) {
	for i := range a.cfg.Tokens {
		token := &a.cfg.Tokens[i]
		if token.Token == raw {
			return token, true
		}
	}
	return nil, false
}

func bearerToken(r *http.Request) string {
	value := r.Header.Get("Authorization")
	if !strings.HasPrefix(value, "Bearer ") {
		return ""
	}
	return strings.TrimSpace(strings.TrimPrefix(value, "Bearer "))
}

func voiceAllowed(token *config.TokenConfig, voice string) bool {
	for _, allowed := range token.AllowedVoices {
		if allowed == voice {
			return true
		}
	}
	return false
}

func proxyAudio(w http.ResponseWriter, resp *http.Response) {
	defer resp.Body.Close()

	w.Header().Set("Content-Type", "audio/mpeg")
	w.WriteHeader(http.StatusOK)
	_, _ = io.Copy(w, resp.Body)
}
```

```go
package httpapi

import (
	"context"
	"encoding/json"
	"net/http"

	"github.com/songtf/tts-stack/gateway/internal/upstream"
)

type speechRequest struct {
	Model string `json:"model"`
	Input string `json:"input"`
	Voice string `json:"voice"`
}

func (a *App) handleSpeech(w http.ResponseWriter, r *http.Request) {
	tokenValue := bearerToken(r)
	if tokenValue == "" {
		writeError(w, http.StatusUnauthorized, "invalid_request_error", "missing bearer token")
		return
	}

	token, ok := a.findToken(tokenValue)
	if !ok || !token.Enabled {
		writeError(w, http.StatusUnauthorized, "invalid_request_error", "invalid bearer token")
		return
	}

	var payload speechRequest
	if err := json.NewDecoder(r.Body).Decode(&payload); err != nil {
		writeError(w, http.StatusBadRequest, "invalid_request_error", "invalid json body")
		return
	}

	if payload.Model != "tts-1" {
		writeError(w, http.StatusBadRequest, "invalid_request_error", "model must be tts-1")
		return
	}

	if payload.Input == "" {
		writeError(w, http.StatusBadRequest, "invalid_request_error", "input is required")
		return
	}

	if len([]rune(payload.Input)) > token.Defaults.MaxTextLength {
		writeError(w, http.StatusBadRequest, "invalid_request_error", "input exceeds max_text_length")
		return
	}

	voice := payload.Voice
	if voice == "" {
		voice = token.Defaults.Voice
	}
	if !voiceAllowed(token, voice) {
		writeError(w, http.StatusBadRequest, "invalid_request_error", "voice not allowed for this token")
		return
	}

	resp, err := a.upstream.Synthesize(context.Background(), upstream.SynthesizeParams{
		Text:        payload.Input,
		Voice:       voice,
		Thread:      token.Defaults.Thread,
		ShardLength: token.Defaults.ShardLength,
	})
	if err != nil {
		writeError(w, http.StatusBadGateway, "bad_gateway", "upstream tts service failed")
		return
	}

	if resp.StatusCode != http.StatusOK {
		resp.Body.Close()
		writeError(w, http.StatusBadGateway, "bad_gateway", "upstream tts service failed")
		return
	}

	proxyAudio(w, resp)
}
```

```go
package httpapi

import (
	"context"
	"encoding/json"
	"net/http"

	"github.com/songtf/tts-stack/gateway/internal/upstream"
)

type voicesResponse struct {
	DefaultVoice string           `json:"default_voice"`
	Voices       []upstream.Voice `json:"voices"`
}

func (a *App) handleVoices(w http.ResponseWriter, r *http.Request) {
	tokenValue := bearerToken(r)
	if tokenValue == "" {
		writeError(w, http.StatusUnauthorized, "invalid_request_error", "missing bearer token")
		return
	}

	token, ok := a.findToken(tokenValue)
	if !ok || !token.Enabled {
		writeError(w, http.StatusUnauthorized, "invalid_request_error", "invalid bearer token")
		return
	}

	voices, err := a.upstream.ListVoices(context.Background())
	if err != nil {
		writeError(w, http.StatusBadGateway, "bad_gateway", "upstream voice catalog failed")
		return
	}

	filtered := make([]upstream.Voice, 0, len(token.AllowedVoices))
	for _, voice := range voices {
		if voiceAllowed(token, voice.ShortName) {
			filtered = append(filtered, voice)
		}
	}

	w.Header().Set("Content-Type", "application/json")
	_ = json.NewEncoder(w).Encode(voicesResponse{
		DefaultVoice: token.Defaults.Voice,
		Voices:       filtered,
	})
}
```

```go
package main

import (
	"context"
	"log/slog"
	"net/http"
	"os"
	"os/signal"
	"strconv"
	"syscall"

	"github.com/songtf/tts-stack/gateway/internal/config"
	"github.com/songtf/tts-stack/gateway/internal/httpapi"
	"github.com/songtf/tts-stack/gateway/internal/upstream"
)

func main() {
	configPath := os.Getenv("CONFIG_PATH")
	if configPath == "" {
		configPath = "/app/config/gateway.yaml"
	}

	logger := slog.New(slog.NewJSONHandler(os.Stdout, nil))

	store, err := config.NewStore(configPath)
	if err != nil {
		logger.Error("load config failed", "error", err)
		os.Exit(1)
	}

	ctx, stop := signal.NotifyContext(context.Background(), syscall.SIGINT, syscall.SIGTERM)
	defer stop()

	if err := config.Watch(ctx, logger, store, configPath); err != nil {
		logger.Error("watch config failed", "error", err)
		os.Exit(1)
	}

	current := store.Current()
	client := upstream.New(current.Upstream.BaseURL, current.Upstream.TimeoutSeconds)
	app := httpapi.New(current, client)

	server := &http.Server{
		Addr:    ":" + strconv.Itoa(current.Server.Port),
		Handler: app,
	}

	logger.Info("gateway starting", "port", current.Server.Port)
	if err := server.ListenAndServe(); err != nil && err != http.ErrServerClosed {
		logger.Error("server exited", "error", err)
		os.Exit(1)
	}
}
```
